	Animate AmmoNoClip Alpha "255"	Linear 0.0 0.0

	RunEvent LevelInitAny 0.0
}

	Animate	AmmoEverywhereDudeDoubleLow	Fgcolor "155 0 0 255" Linear 0.15 0.06
	
	RunEvent AmmoLowPulseLoopb	0.3
}

}
